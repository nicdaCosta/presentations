Detecting An Array

  var coffeeScriptBro = [],
      isArray = false;

  // Duck Typing
  if ( 'splice' in coffeeScriptBro && 'join' in coffeeScriptBro ) {
    isArray = true;
  }

  // Comparing Internal Class Property
  if ( Object.prototype.toString.call( coffeeScriptBro ) === '[object Array]' ) {
    isArray = true;
  }
  
  // ES5
  isArray = Array.isArray( coffeeScriptBro );

Detecting If A Value Is In An Array


  var progLanguages = [ 'Python' , 'Ruby' , 'C++' , 'JavaScript' , 'C#' , 'PHP' ],
      whatIKnow = 'JavaScript',
      index;

  // ES3
  for ( var i = 0; i < progLanguages.length; i++) {
    if ( whatIKnow === progLanguages[ i ] ) {
      index = i;
      break;
    }
  }

  // ES5
  index = progLanguages.indexOf( whatIKnow );

Make Objects Observable

Snippet By Martin Cronjé - @martincronje

Source: jsfiddle.net/martincronje/Dz9WD/


 function makePropertyObservable(obj, prop, fn) {
    var oldDesc = Object.getOwnPropertyDescriptor(obj, prop);

    if (!oldDesc.get) oldDesc = makePropertyDescriptor(oldDesc);

    var newDesc = {
        set: function (newValue) {
            var diff = {
                name: prop,
                oldValue: oldDesc.get.call(this),
                newValue: newValue
            };
            fn(this, diff)
            oldDesc.set.call(this, newValue);
        }
    };
    Object.defineProperty(obj, prop, newDesc);
}
